calcular_pesos <- function(data, var){
  data  %>%
  select(ID_Clone, Trait, var) %>%
  pivot_wider(names_from = Trait, values_from = var) %>%
  mutate(
    N_Roots = 15 * N_Roots,
    FRY = 20 * FRY,
    ShY = 10 * ShY,
    DMC = 15 * DMC,
    StY = 10 * StY,
    Plant.Height = 5 * Plant.Height,
    HI = 10 * HI,
    StC = 10 * StC,
    Root.Le = 5 * Root.Le,
    Root.Di = 5 * Root.Di,
    Stem.D = 5 * Stem.D,
    Nstem.Plant = 5 * Nstem.Plant
  ) %>%
  mutate(pesos =
           rowSums(.[2:13], na.rm = TRUE))
}
